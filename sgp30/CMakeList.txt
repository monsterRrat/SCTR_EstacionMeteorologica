cmake_minimum_required(VERSION 3.15)
include(pico_sdk_import.cmake)

# Definir el nombre del proyecto
project(SCTR_EstacionMeteorologica
	LANGUAGES CXX C ASM
)

# Inicializar el SDK de Pico
pico_sdk_init()

# Añadir el ejecutable para el sensor sgp30
add_executable(sgp30
    driver_sgp30.c
)


# Enlazar las bibliotecas necesarias
target_link_libraries(sgp30
    pico_stdlib    # Librería estándar de Pico
    hardware_i2c   # Para trabajar con I2C
    hardware_adc   # Para trabajar con el ADC
)

target_include_directories(sgp30 PUBLIC
    ${CMAKE_CURRENT_LIST_DIR}
)

# Habilitar/Deshabilitar la salida por USB y UART
pico_enable_stdio_usb(sgp30 1)
pico_enable_stdio_uart(sgp30 0)

# Crear archivos adicionales (binario, map, hex, uf2)
pico_add_extra_outputs(sgp30)
